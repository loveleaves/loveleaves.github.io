def solveNQueens(self, n: int) -> List[List[str]]:

def coinChange(self, coins: List[int], amount: int) -> int:
    @cache
    def dfs(i, c):
        if i<0:
            if c==0:
                return 0
            return inf
        if coins[i] > c:
            return dfs(i-1, c)
        return min(dfs(i-1, c), dfs(i, c-coins[i])+1)
    ans = dfs(len(coins)-1, amount)
    return ans if ans < inf else -1
